MySQL
cmd环境下：
	启动服务：
		net start mysql
	停止服务：
		net stop mysql
		
	打开mysql:
		mysql -x		x 为不同字母选项
		
	退出：
		mysql > quit;
		mysql > exit;
		mysql > \q;
		
修改mysql提示符：
	连接客户端时通过参数设定：
		shell>mysql -uroot -proot --prompt 提示符
		
	连接上客户端后，通过prommt命令修改：
		mysql > prompt 提示符
		
	mysql提示符:
		\D	完整的日期
		\d	当前数据库
		\h	服务器名称
		\u	当前用户
		
mysql语句的规范：
		关键字与函数名称全部大写
		数据库名称、表名称、字段名称全部小写
		SQL语句必须以分号结尾
		（以上只是规范，实际使用中大小写没有限制）
mysql 常用命令：
		显示当前服务器版本
		SELECT VERSION();
		显示当前日期时间
		SELECT NOW();
		显示当前用户
		SELECT USER();
		
数据库的操作：
	创建数据库：
		CREATE {DATABASE|SCHEMA}[IF NOT EXISTS] db_name [DEFAULT]CHARACTER SET [=] charset_name
		{}中的元素是必须有的，[]中的是可有可无的，|表示两个元素中选一个
		
	查看当前服务器下的数据表列表：
		SHOW {DATABASE | SCHEMA} [LIKE 'pattern'|WHERE expr]
		
	显示错误：
		SHOW WARNINGS
		SHOW CREATE DATABASE test1；
	修改数据库：
		ALTER {DATABASE | SCHEMA} [db_name] [DEFAULT] CHARACTER SET [=] character_name
	删除数据库：
		DROP {DATABASE | SCHEMA} [IF EXISTS] db_name
		例子：
			DROP DATABASE　db_name;
			
MYSQL 默认端口号 3306
mysql 中的超级用户 root
创建数据库	CREATE DATABASE
修改数据库	ALTER DATABASE
删除数据库	DROP DATABASE
